・サイト構成のポイント

まず外側を全体で包むことが基本になっています！
なので↓のようなイメージで作成していきます


//省略
</head>
<body>
  <div class="wrap">

    <header id="header">
          //この中に書いていく
    </header>
    <!-- /#header -->

    <div class="mainvisual">
          //この中に書いていく
    </div>
    <!-- /.mainvisual -->

    <div class="content">
          //この中に書いていく
    </div>
    <!-- /.content -->


    <div id="footer">
          //この中に書いていく
    </div>
    <!-- /#footer -->
  </div>
</body>
</html>
